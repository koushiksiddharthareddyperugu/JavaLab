public class MultiplicationTables {
    public static void main(String[] args) {

        System.out.println("Table of 3:");
        for(int i = 1; i <= 10; i++) {
            System.out.println("3 x " + i + " = " + (3 * i));
        }

        System.out.println("\nTable of 5:");
        for(int i = 1; i <= 10; i++) {
            System.out.println("5 x " + i + " = " + (5 * i));
        }
    }
}
